快速开始

3 分钟上手，选择适合你的方式

bash
# 1. 安装（需要 Python 3.9+）
pip3 install oh-my-coder

# 1.1 ⚠️ 配置 PATH（否则 omc 命令找不到）
# 将以下行加入 ~/.zshrc（macOS）或 ~/.bashrc（Linux）：
export PATH="$HOME/Library/Python/3.9/bin:$PATH"
source ~/.zshrc

# 2. 验证安装
omc --version

# 3. 配置 API Key（免费模型 GLM-4-Flash）
# 获取 Key：https://open.bigmodel.cn → 注册 → API Keys → 创建
omc config set glm_api_key "your-api-key"
omc config set default_model glm-4-flash

# 4. 开始使用！直接输入需求即可
omc run "帮我分析这个项目的架构"

# 更多命令：
# omc models list 查看可用模型
# omc review --path ./src 代码审查
# omc doctor 检查环境
bash
# 1. 克隆项目（Web 界面需从源码运行）
git clone https://github.com/vobc/oh-my-coder.git
cd oh-my-coder

# 2. 安装依赖（需要 Python 3.9+）
pip3 install -e ".[web]"

# 3. 启动 Web 服务（自动打开浏览器）
python3 -m src.web.app
# 浏览器访问 http://localhost:8000

# 4. 在网页中输入任务，点击执行
# 例如：「帮我写一个用户登录功能」

# 提示：Web 界面支持 9 个模型可切换
bash
# 终端窗口 1 - 启动后端 API（保持运行）
pip3 install oh-my-coder

# ⚠️ 配置 PATH（如果 omc 命令找不到）：
export PATH="$HOME/Library/Python/3.9/bin:$PATH"

omc server start
# 看到 "Uvicorn running on http://127.0.0.1:8080" 即成功

# 终端窗口 2 - 启动桌面窗口
git clone https://github.com/vobc/oh-my-coder.git
cd oh-my-coder/desktop
npm install
npm run electron:dev
# 提示：桌面端支持多 Agent 协作可视化
